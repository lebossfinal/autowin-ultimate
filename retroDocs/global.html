    console.log(`Ally can still act`);

    await spell({id: 123, cell: enemyCell});
} else {
    console.log("Not enough action points for this spell");
}
// Plan combat actions based on available AP
const currentAP = getActionPoints();
if (currentAP >= 6) {
    // Can cast expensive spell
    await spell({id: powerfulSpellId, cell: enemyCell});
} else if (currentAP >= 3) {
    // Use medium cost spell
    await spell({id: mediumSpellId, cell: enemyCell});
} else if (currentAP >= 1) {
    // Use basic attack
    await spell({id: basicAttackId, cell: enemyCell});
} else {
    // Pass turn if no AP left
    await passTurn();
}
// Optimize spell sequence based on AP
let remainingAP = getActionPoints();
const spells = [
    {id: 123, cost: 4, priority: 'high'},
    {id: 456, cost: 3, priority: 'medium'},
    {id: 789, cost: 2, priority: 'low'}
];

for (const spell of spells) {
    if (remainingAP >= spell.cost) {
        await spell({id: spell.id, cell: enemyCell});
        remainingAP -= spell.cost;
    }
}

hasCooldown(args) → {boolean|undefined}

Description:

Examples

}

    {id: 123, cooldown: 4}, // Powerful spell, 4-turn cooldown
    {id: 456, cooldown: 2}, // Medium spell, 2-turn cooldown
    {id: 789, cooldown: 1}  // Basic spell, 1-turn cooldown
];

for (const spellData of spellRotation) {
    if (hasCooldown(spellData) && getActionPoints() >= getSpellCost(spellData.id)) {
        await spell({id: spellData.id, cell: enemyCell});
        break; // Cast first available spell
    }
}
// Check multiple spells and choose the best available

    {id: 111, cooldown: 5, damage: 'high'},
    {id: 222, cooldown: 3, damage: 'medium'},
    {id: 333, cooldown: 1, damage: 'low'}
].filter(spell => hasCooldown(spell));

if (availableSpells.length > 0) {
    const bestSpell = availableSpells[0]; // Use highest damage available
    await spell({id: bestSpell.id, cell: enemyCell});
}
// Safe cooldown check with error handling
try {
    if (hasCooldown({id: 456, cooldown: 3})) {
        await spell({id: 456, cell: targetCell});

    console.error("Spell check failed:", error.message);

// Conditional casting based on combat state
if (isInFight() && turn > 0) {
    const canCastHeal = hasCooldown({id: healSpellId, cooldown: 4});
    const canCastAttack = hasCooldown({id: attackSpellId, cooldown: 2});

    if (myHealth < 50 && canCastHeal) {
        await spell({id: healSpellId, cell: myCell()});
    } else if (canCastAttack) {
        await spell({id: attackSpellId, cell: enemyCell});
    }
}
// Track cooldowns for strategic planning

    {id: 100, cooldown: 6, name: "Ultimate Attack"},
    {id: 200, cooldown: 4, name: "Shield"},
    {id: 300, cooldown: 3, name: "Heal"}
];

importantSpells.forEach(spell => {
    const available = hasCooldown(spell);
    console.log(`${spell.name}: ${available ? 'Available' : 'On Cooldown'}`);
});
